# Tabs

Tabs låter användarna navigera mellan relaterade delar av innehållet, där en del visas åt gången.

<Primary />
<Controls />

## Användning

```tsx
import { Tabs } from '@digdir/designsystemet-react';

<Tabs defaultValue='value1'>
  <Tabs.List>
    <Tabs.Tab value='value1'>Flik 1</Tabs.Tab>
    <Tabs.Tab value='value2'>Flik 2</Tabs.Tab>
    <Tabs.Tab value='value3'>Flik 3</Tabs.Tab>
  </Tabs.List>
  <Tabs.Panel value='value1'>innehåll 1</Tabs.Panel>
  <Tabs.Panel value='value2'>innehåll 2</Tabs.Panel>
  <Tabs.Panel value='value3'>innehåll 3</Tabs.Panel>
</Tabs>;
```
